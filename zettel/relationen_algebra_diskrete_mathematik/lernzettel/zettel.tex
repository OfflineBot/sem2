\subsection{Grundlagen}

\begin{itemize}
  \item \textbf{Logik}: Wissen darstellen, Schlussfolgerungen ziehen. Anwendung: Hardware, Programmiersprachen, DBs, Algorithmen, ML.
  \item \textbf{Atomare Aussage}: Satz, eindeutig wahr (w) oder falsch (f). VerknГјpfbar mit вҲ§, вҲЁ, В¬.
  \item \textbf{Aussagenvariablen}: a, b, c, вҖҰ (nicht w/f).
  \item \textbf{Belegung}: Zuordnung von Wahrheitswerten zu Variablen.
\end{itemize}

\subsubsection{Verfahren kDN/kKN}

\subsection{Resolution}

\begin{itemize}
  \item \textbf{Beweis durch Widerspruch}: zeige (AвӮҒ вҲ§ вҖҰ вҲ§ AвӮҷ вҲ§ В¬b) widerspruchsvoll
  \item Voraussetzung: KN

  \item KN вҶ’ Klauselmenge: вҲ§ und вҲЁ entfernen, Mengen-Notation

  \item \textbf{Leere Klausel} в–Ў: Widerspruch bewiesen
  \item Pro Schritt nur EIN Literal entfernen

  \item Verwendete Klauseln bleiben in Menge

\subsubsection{\texttt{dijkstra(graph, start, end=FALSE)}}
\begin{lstlisting}

distances[start]=0; previous[start]=0; current=start
while(any(!visited)):
  visited[current]=TRUE
  if(current==end) break
  for i Nachbarn:
    if graph[current,i]!=0 && !visited[i] && distances[i] > distances[current]+graph[current,i]:
      distances[i]=distances[current]+graph[current,i]
      previous[i]=current
  current = unvisited Knoten mit kleinstem distances; sonst current=-1
\end{lstlisting}

\begin{itemize}
  \item \texttt{end=FALSE}: kГјrzeste Pfade zu allen Knoten
\end{itemize}


\subsubsection{\texttt{A*(graph, start, end, heuristic)}}
\begin{itemize}

  \item Update setzt zusГӨtzlich \texttt{priorities[i] = distances[i] + heuristic[i]}
  \item Knotenwahl: \texttt{min priorities[i]} Гјber \texttt{!visited[i]}
\end{itemize}

\section{TSP вҖ“ Travelling Salesperson Problem}

\textbf{Definition}: n StГӨdte alle genau einmal besuchen, zur Startstadt zurГјck, minimale TourlГӨnge.

\begin{itemize}
  \item Graph: V=StГӨdte, E=Strecken, W=LГӨngen
\end{itemize}


\subsection{Anwendungen}
\begin{itemize}
  \item Logistik (VRP-Erweiterungen)
  \item Mikrochip-Design, Leiterplattenbohrung
  \item Genom-Sequenzierung (Knoten = DNA-TeilstrГӨnge)
\end{itemize}


\subsection{Verwandte Probleme}
\begin{itemize}
  \item TSP ohne RГјckkehr
  \item VRP (mehrere Reisende)
  \item Traveling Thief (TSP + Knapsack)
  \item Stochastisch / dynamisch
\end{itemize}


\subsection{LГ¶sungsverfahren}

\textbf{Brute Force}: alle Touren durchprobieren вҖ” optimal, exponentielle Laufzeit.


\textbf{Nearest Neighbour Heuristik (NNA)}: Greedy-Verfahren ausgehend vom Startknoten.

\begin{enumerate}
  \item Start bei Startknoten
  \item NГӨchster Knoten: gГјnstigste Kante zu unbesuchtem Knoten
  \item Wiederhole bis alle besucht
  \item StartknotenabhГӨngig, meist suboptimal
  \item Verbesserung: jeden Knoten als Start probieren вҶ’ bestes Ergebnis (Laufzeit Г—n, weiterhin keine Garantie)
\end{enumerate}

\textbf{2-Opt Heuristik}: Lokale Suche zur Verbesserung einer vorhandenen Tour.

\begin{enumerate}
  \item Approximative StartlГ¶sung (z. B. NNA)
  \item Iterativ TeilstГјck invertieren, falls neue Tour kГјrzer вҶ’ Гјbernehmen
  \item NГӨchster Tausch
  \item Approximativ
\end{enumerate}
